//	ignoring labels
LABEL_IN_MAIN_CB ::= LABEL

//	normal xml miniparsers


LABEL ::= _ <  labelname  > _ LABELS _ </  labelname  >                               ##transf2-> $(__nothing__)


LABEL ::= _ < labelname spcs PROPERTIES_INLABEL spcs />                               ##transf2-> $(__nothing__)
LABEL ::= _ < labelname spcs PROPERTIES_INLABEL _ > _ LABELS _ </ labelname  >        ##transf2-> $(__nothing__)
LABEL ::= _ < labelname spcs />                                                       ##transf2-> $(__nothing__)

PROPERTIES_INLABEL ::=  prop_name  =  " PROP_VALUE "   spcs PROPERTIES_INLABEL        ##transf2-> $(__nothing__)
PROPERTIES_INLABEL ::=  prop_name  =  " PROP_VALUE "                                  ##transf2-> $(__nothing__)


prop_name ::= ([a-zA-Z0-9_]+)

PROP_VALUE ::= propchars  PROP_VALUE
PROP_VALUE ::= propchars 

propchars  ::= ([a-zA-Z0-9 _\\\.\-\:]+)
propchars  ::= (\/[^>])



labelname ::= ([a-zA-Z0-9_]+)

" ::= (\")
= ::= (\=)
< ::= (\<)
> ::= (\>)
</ ::= (\<\/)
/> ::= (\/\>)


_ ::= ([ \r\n\t]*)
spcs ::=  ([ \n\r\t]+)
